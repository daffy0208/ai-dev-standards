export async function syncCustomers() {
  // Extract
  const raw = await fetchFromAPI()

  // Transform
  const validated = raw.map(validateCustomer)
  const enriched = await Promise.all(validated.map(enrichCustomer))

  // Load
  await batchUpsert(enriched)

  // Monitor
